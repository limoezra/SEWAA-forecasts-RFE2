View forecasts API version 4.0

To use the functions documented here, reference the file plotForecastsMin.js in your HTML. 

	<script src="plotForecastsMin.js"></script>

The methods for loading and drawing a forecast to an html canvas are defined in a countsData object.

countsData methods are usually asynchronous. 

	// Create a countsData object
	let GANForecast = new countsData();

Methods:

loadGANForecast
regionSpec.loadRegionNames
computeLatLonIdxBounds
plotExceedanceProbability
plotExceedanceValue
plotMean
plotStd
plotHistogram

countsData.loadGANForecast

countsData.loadGANForecast(fileName, modelName, accumulationHours)

Loads a forecast for a particular initialisation date and valid time.

Arguments

fileName : A string containing the forecast data file name including the relative path.
modelName : Can currently be either "6h accumulation" or "24h accumulation".
accumulationHours : Can currently be either 6 or 24, and must correspond to the correct model name. 

	// Create a countsData object
	let GANForecast = new countsData();
	
	// Load a forecast
	let fileName = "PathToFirstForecastFile";
	let modelName = "6h accumulation";
	let accumulationHours = 6;
	await GANForecast.loadGANForecast(fileName, modelName, accumulationHours);
	
	// Do some processing/plotting
	// ...
	
	// Load another forecast
	fileName = "PathToSecondForecastFile";
	modelName = "24h accumulation";
	accumulationHours = 24;
	await GANForecast.loadGANForecast(fileName, modelName, accumulationHours);

countsData.regionSpec.loadRegionNames

countsData.regionSpec.loadRegionNames(path, fileName)

Load all of the country and region names and the country bounding latitude and longitude box from a json file. Currently the region names are unused.

Arguments

path : Relative path to the json file.
fileName : File name of the json file. 

	// Create a countsData object
	let GANForecast = new countsData();
	
	// Load the region names
	await GANForecast.regionSpec.loadRegionNames("../boundaries", "regional_names.json");

countsData.computeLatLonIdxBounds

countsData.computeLatLonIdxBounds(regionName)

Arguments

regionName : Regions can currently be "Burundi", "Djibouti", "Eritrea", "Ethiopia", "Kenya", "Rwanda", "Somalia", "South Sudan", "Sudan", "Tanzania", "Uganda", "ICPAC", "East Africa", "All".

Returns

An array of four numbers in the form [minimumLatitudeIndex, maximumLatitudeIndex, minimumLongitudeIndex, maximumLongitudeIndex] corresponding to the index of the grid locations for that region name. 

	// Create a countsData object
	let GANForecast = new countsData();
	
	// Load the region names
	await GANForecast.regionSpec.loadRegionNames("../boundaries", "regional_names.json");
	
	// Compute the latitude and longitude index boundaries
	let regionName = "East Africa";
	let [minLatIdx, maxLatIdx, minLonIdx, maxLonIdx] = GANForecast.computeLatLonIdxBounds(regionName);

countsData.plotExceedanceProbability

countsData.plotExceedanceProbability(ctx, x, y, width, height, maxRain, units, style, showPercentages, regionName)

Draw a map of the probability of exceeding a given rainfall threshold at each location.

Arguments

ctx : The html canvas to draw to.
x : Pixels from the left of the canvas where the plot starts.
y : Pixels from the top of the canvas where the plot starts.
width : Width of the plot in pixels including the heading and legend etc.
height : Height of the plot in pixels including the heading and legend etc.
maxRain : Rainfall threshold in mm/h.
units : The units to use in the plot. Can be "mm/h", "mm/6h", "mm/day", "mm/week".
style : The plot style. Can be "Default", "ICPAC", "KMD", "EMI".
showPercentages : Either true to display a legend with percentages, or false to display a legend with words.
regionName : Regions can currently be "Burundi", "Djibouti", "Eritrea", "Ethiopia", "Kenya", "Rwanda", "Somalia", "South Sudan", "Sudan", "Tanzania", "Uganda", "ICPAC", "East Africa", "All".

Returns

An array of four numbers in the form [xMin, yMin, xMax, yMax] corresponding to the canvas rectangle within which the exceedance probability is drawn. Used for example to detect where on the map the user has clicked. 

	// Create a countsData object
	let GANForecast = new countsData();
	
	// Load a forecast
	let fileName = "PathToForecastFile";
	let modelName = "6h accumulation";
	let accumulationHours = 6;
	await GANForecast.loadGANForecast(fileName, modelName, accumulationHours);
	
	// Find the context to draw to
	const canvas = document.getElementById("myCanvas");
	const ctx = canvas.getContext("2d");
	
	// Draw the exceedance probabilities
	let x = 5;
	let y = 10;
	let width = 500;
	let height = 500;
	let maxRain = 1/24;  // Rainfall threshold in mm/h
	let units = "mm/6h";
	let style = "Default";
	let showPercentages = true;
	let regionName = "East Africa";
	let plotRect = await GANForecast.plotExceedanceProbability(ctx, x, y, width, height, maxRain, units, style, showPercentages, regionName);

countsData.plotHistogram

countsData.plotHistogram(ctx, x, y, width, height, maxRain, probability, latitudeIdx, longitudeIdx, units, barChartSpec)

Draw a histogram of the rainfall predicted to fall at a particular latitude and longitude grid box.

Arguments

ctx : The html canvas to draw to.
x : Pixels from the left of the canvas where the plot starts.
y : Pixels from the top of the canvas where the plot starts.
width : Width of the plot in pixels including the heading and legend etc.
height : Height of the plot in pixels including the heading and legend etc.
maxRain : Rainfall threshold in mm/h.
probability : Probability threshold between 0 and 1. The plotted rainfall value corresponds to the maximum rainfall with the given probability.
latitudeIdx : The grid cell latitude index to use for the plot.
longitudeIdx : The grid cell longitude index to use for the plot.
units : The units to use in the plot. Can be "mm/h", "mm/6h", "mm/day", "mm/week".
barChartSpec : Depreciated, will be removed. A data structure containing the bar chart specification. Create with let barChartSpec = new barChartSpecification(); 

	// Create a countsData object
	let GANForecast = new countsData();
	
	// Load a forecast
	let fileName = "PathToForecastFile";
	let modelName = "6h accumulation";
	let accumulationHours = 6;
	await GANForecast.loadGANForecast(fileName, modelName, accumulationHours);
	
	// Find the context to draw to
	const canvas = document.getElementById("myCanvas");
	const ctx = canvas.getContext("2d");
	
	// Draw the histogram
	let x = 5;
	let y = 10;
	let width = 500;
	let height = 500;
	let maxRain = 1/24;  // Rainfall threshold in mm/h
	let probability = 0.95;
	let latitudeIdx = 120;
	let longitudeIdx = 50;
	let units = "mm/6h";
	let barChartSpec = new barChartSpecification();  // Depreciated, will be removed.
	await GANForecast.plotHistogram(ctx, x, y, width, height, maxRain, probability, latitudeIdx, longitudeIdx, units, barChartSpec);
